# Documentation site toolchain (mkdocs.yml). PINNED so the CI build
# (.github/workflows/docs.yml) and a local build can never drift.
#
#   python3 -m venv .venv
#   .venv/bin/pip install -r docs/requirements.txt
#   .venv/bin/mkdocs build --strict
#
# --strict is load-bearing: it promotes MkDocs warnings to errors, so a renamed or
# deleted page that leaves a dangling link fails the build instead of shipping a 404.

mkdocs-material==9.7.6
pymdown-extensions==11.0.1
